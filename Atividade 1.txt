0000-EEXXNN02.
	PERFORM 1000-INICIALIZAR
	PERFORM 2000-PROCESSSAR 
			UNTIL WS-FIM = "S"
	PERFORM 3000-TERMINO 
	STOP RUN
	.
1000-INICIALIZAR.
	MOVE ZERO TO WS-ACUMULADOR 
				 WS-CTHOSPITAL
	MOVE "N"  TO WS-FIM
	PERFORM 1500-LER-SYSIN
	.
1500-LER-SYSIN.
	ACCEPT WS-QTDPACIENTES FROM SYSIN
	IF WS-QTDPACIENTES = 0 
		MOVE "S" TO WS-FIM 
	ELSE 
		ADD 1 TO WS-CTHOSPITAL
	END-IF 
	.
2000-PROCESSAR.
	ADD WS-QTDPACIENTES TO WS-QTDPACIENTES
	PERFORM 1500-LER-SYSIN
	.
3000-TERMINO.
	DISPLAY "QUANTIDADE HOSPITAIS PROCESSADOS = " 
									WS-CTHOSPITAL
	DISPLAY "QUANTIDADE TOTAL DE PACIENTES = " 
									WS-ACUMULADOR
	DISPLAY "TERMINO NORMAL DO EEXXNN02"
	.
